class Status:
    """A regular Cryostream status packet."""

    type_status: int
    gas_set_point: int
    gas_temp: int
    gas_error: int
    run_mode: int
    phase_id: int
    ramp_rate: int
    target_temp: int
    evap_temp: int
    suct_temp: int
    remaining: int
    gas_flow: int
    gas_heat: int
    evap_heat: int
    suct_heat: int
    line_pressure: int
    alarm_code: int
    run_time: int
    controller_number: int
    software_version: int
    evap_adjust: int
    status_bytes_string: str = ">BBHHhBBHHHHHBBBBBBHHBB"

    def from_packed(cls, b: bytes) -> "Status":
        """Create a status packet from its packed byte format."""
        return cls(*struct.unpack(cls.status_bytes_string, b))

            self.controller_number,
            self.software_version,
            self.evap_adjust,
        )

        return status_bytes
